import neoapi
cam = neoapi.Cam() # create a camera object
cam.Connect() # connect to any device
# the internal list is accessed only one time to get the iterator
feat = cam.GetFeatureList().find("SpecialFeature");
if feat.GetName(): # check if feature exists
feat.__ref__().SetInt(1) # set feature
# or (every time the feature name is passed, a search on the internal list takes place.)
if cam.HasFeature("SpecialFeature"): # check if feature exists
cam.SetFeature("SpecialFeature", 1) # set feature
