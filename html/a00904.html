60  def Detect(self, image, show_image):
61  cambuf = image.GetUserBuffer()
62  img_mat = cambuf.cpu_mat if cambuf.memtype == MemoryMode.cpu else cv2.UMat(cambuf.cpu_mat)
63  if neoapi.PixelFormat_BayerRG8 == self._pixelformat:
64  self._grey_mat = cv2.cvtColor(img_mat, cv2.COLOR_BayerRG2GRAY)
65  else:
66  self._grey_mat = img_mat
67  self._gauss_mat = cv2.GaussianBlur(self._grey_mat, (5, 5), 0)
68  self._sobel_mat = cv2.Sobel(self._gauss_mat, 0, 1, 1, ksize=5)
69  if show_image:
70  cv2.imshow(self._identifier, self._sobel_mat)
