61  def Detect(self, image, show_image):
62  cambuf = image.GetUserBuffer()
63  img_mat = cambuf.cpu_mat if cambuf.memtype == MemoryMode.cpu else cv2.UMat(cambuf.cpu_mat)
64  if neoapi.PixelFormat_BayerRG8 == self._pixelformat:
65  self._grey_mat = cv2.cvtColor(img_mat, cv2.COLOR_BayerRG2GRAY)
66  else:
67  self._grey_mat = img_mat
68  self._gauss_mat = cv2.GaussianBlur(self._grey_mat, (5, 5), 0)
69  self._sobel_mat = cv2.Sobel(self._gauss_mat, 0, 1, 1, ksize=5)
